105void MemaReComponent::setFaderbankCtrlActive()
106{
107 auto resizeRequired = false;
108
109 if (m_faderbankCtrlComponent)
110 {
111 if (!m_faderbankCtrlComponent->isVisible())
112 {
113 m_faderbankCtrlComponent->setIOCount(m_currentIOCount);
114 m_faderbankCtrlComponent->setVisible(true);
115 resizeRequired = true;
116 }
117 }
118 if (m_panningCtrlComponent && m_panningCtrlComponent->isVisible())
119 {
120 m_panningCtrlComponent->resetCtrl();
121 m_panningCtrlComponent->setVisible(false);
122 resizeRequired = true;
123 }
124 if (m_pluginCtrlComponent && m_pluginCtrlComponent->isVisible())
125 {
126 m_pluginCtrlComponent->resetCtrl();
127 m_pluginCtrlComponent->setVisible(false);
128 resizeRequired = true;
129 }
130
131 if (resizeRequired && !getLocalBounds().isEmpty())
132 resized();
133}

269void MemaReComponent::handleMessage(const Message& message)
270{
271 if (RunningStatus::Active != m_runningStatus)
272 {
273 m_runningStatus = RunningStatus::Active;
274 resized();
275 }
276
277 if (auto const apm = dynamic_cast<const Mema::AnalyzerParametersMessage*>(&message))
278 {
279 // we don't want analyzer parameter infos!
280 DBG(juce::String(__FUNCTION__) + " ignoring unexpected AnalyzerParametersMessage...");
281 }
282 else if (auto m = dynamic_cast<const Mema::AudioBufferMessage*>(&message))
283 {
284 // we don't want audio buffer infos!
285 DBG(juce::String(__FUNCTION__) + " ignoring unexpected AudiBufferMessage...");
286 }
287 else if (auto const iom = dynamic_cast<const Mema::ReinitIOCountMessage*>(&message))
288 {
289 DBG(juce::String(__FUNCTION__) + " handling ReinitIOCountMessage...");
290
291 auto inputCount = iom->getInputCount();
292 jassert(inputCount > 0);
293 auto outputCount = iom->getOutputCount();
294 jassert(outputCount > 0);
295
296 m_currentIOCount = std::make_pair(inputCount, outputCount);
297
298 if (m_faderbankCtrlComponent)
299 m_faderbankCtrlComponent->setIOCount(m_currentIOCount);
300 if (m_panningCtrlComponent)
301 m_panningCtrlComponent->setIOCount(m_currentIOCount);
302
303 resized();
304 }
305 else if (auto const cpm = dynamic_cast<const Mema::ControlParametersMessage*>(&message))
306 {
307 DBG(juce::String(__FUNCTION__) + " handling ControlParametersMessage...");
308
309 for (auto const& inputMuteState : cpm->getInputMuteStates())
310 m_inputMuteStates[inputMuteState.first] = inputMuteState.second;
311 if (!m_inputMuteStates.empty())
312 {
313 if (m_faderbankCtrlComponent)
314 m_faderbankCtrlComponent->setInputMuteStates(m_inputMuteStates);
315 if (m_panningCtrlComponent)
316 m_panningCtrlComponent->setInputMuteStates(m_inputMuteStates);
317 }
318
319 for (auto const& outputMuteState : cpm->getOutputMuteStates())
320 m_outputMuteStates[outputMuteState.first] = outputMuteState.second;
321 if (!m_outputMuteStates.empty())
322 {
323 if (m_faderbankCtrlComponent)
324 m_faderbankCtrlComponent->setOutputMuteStates(m_outputMuteStates);
325 if (m_panningCtrlComponent)
326 m_panningCtrlComponent->setOutputMuteStates(m_outputMuteStates);
327 }
328
329 for (auto const& cpsIKV : cpm->getCrosspointStates())
330 for (auto const& cpsOKV : cpsIKV.second)
331 m_crosspointStates[cpsIKV.first][cpsOKV.first] = cpsOKV.second;
332 if (!m_crosspointStates.empty())
333 {
334 if (m_faderbankCtrlComponent)
335 m_faderbankCtrlComponent->setCrosspointStates(m_crosspointStates);
336 if (m_panningCtrlComponent)
337 m_panningCtrlComponent->setCrosspointStates(m_crosspointStates);
338 }
339
340 for (auto const& cpvIKV : cpm->getCrosspointValues())
341 for (auto const& cpvOKV : cpvIKV.second)
342 m_crosspointValues[cpvIKV.first][cpvOKV.first] = cpvOKV.second;
343 if (!m_crosspointValues.empty())
344 {
345 if (m_faderbankCtrlComponent)
346 m_faderbankCtrlComponent->setCrosspointValues(m_crosspointValues);
347 if (m_panningCtrlComponent)
348 m_panningCtrlComponent->setCrosspointValues(m_crosspointValues);
349 }
350
351 resized();
352 }
353 else if (auto const ppim = dynamic_cast<const Mema::PluginParameterInfosMessage*>(&message))
354 {
355 DBG(juce::String(__FUNCTION__) + " handling PluginParameterInfosMessage (" + juce::String(ppim->getParameterInfos().size()) + ") ...");
356
357 m_pluginName = ppim->getPluginName().toStdString();
358 m_pluginParameterInfos = ppim->getParameterInfos();
359
360 if (m_pluginCtrlComponent && m_pluginCtrlComponent->isVisible())
361 {
362 m_pluginCtrlComponent->setPluginName(m_pluginName);
363 m_pluginCtrlComponent->setParameterInfos(m_pluginParameterInfos);
364 }
365
366 resized();
367 }
368 else if (auto const ppvm = dynamic_cast<const Mema::PluginParameterValueMessage*>(&message))
369 {
370 DBG(juce::String(__FUNCTION__) + " handling PluginParameterValueMessage (" + juce::String(ppvm->getParameterIndex()) + "; " + ppvm->getParameterId() + "; " + juce::String(ppvm->getCurrentValue()) + ") ...");
371
372 auto paramIdx = ppvm->getParameterIndex();
373 if (paramIdx < m_pluginParameterInfos.size())
374 {
375 m_pluginParameterInfos[paramIdx].currentValue = ppvm->getCurrentValue();
376 m_pluginParameterInfos[paramIdx].id = ppvm->getParameterId();
377 }
378
379 if (m_pluginCtrlComponent && m_pluginCtrlComponent->isVisible())
380 m_pluginCtrlComponent->setParameterValue(paramIdx, ppvm->getParameterId().toStdString(), ppvm->getCurrentValue());
381
382 resized();
383 }
384}
