31 struct LevelVal
32 {
33 LevelVal()
34 {
35 peak = 0.0f;
36 rms = 0.0f;
37 hold = 0.0f;
38 peakdB = 0.0f;
39 rmsdB = 0.0f;
40 holddB = 0.0f;
41 minusInfdb = -10000.0f;
42 }
43 LevelVal(float p, float r, float h, float infdb = -100.0f)
44 {
45 peak = p;
46 rms = r;
47 hold = h;
48 peakdB = juce::Decibels::gainToDecibels(peak, infdb);
49 rmsdB = juce::Decibels::gainToDecibels(rms, infdb);
50 holddB = juce::Decibels::gainToDecibels(hold, infdb);
51 minusInfdb = infdb;
52 }
53
54 float GetFactorRMSdB()
55 {
56 return (-1 * minusInfdb + rmsdB) / (-1 * minusInfdb);
57 }
58 float GetFactorPEAKdB()
59 {
60 return (-1 * minusInfdb + peakdB) / (-1 * minusInfdb);
61 }
62 float GetFactorHOLDdB()
63 {
64 return (-1 * minusInfdb + holddB) / (-1 * minusInfdb);
65 }
66
67 float peak;
68 float rms;
69 float hold;
70 float peakdB;
71 float rmsdB;
72 float holddB;
73 float minusInfdb;
74 };
